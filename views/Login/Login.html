from js import document, location from pyodide import ffi # 로그인 버튼 클릭 시 호출되는 함수 def doLogin(args): print('Do Login ...') # 회원가입 버튼 클릭 시 호출되는 함수 def goToSignUp(args): location.href = '/views/SignUp/SignUp.html'; # 로그인 버튼에 doLogin 함수를 연결해주는 작업 doLoginProxy = ffi.create_proxy(doLogin) document.getElementById('loginBtn').addEventListener('click', doLoginProxy) # 회원가입 버튼에 goToSignUp 함수를 연결해주는 작업 doGoToSignUpProxy = ffi.create_proxy(goToSignUp) document.getElementById('goToSignUp').addEventListener('click', doGoToSignUpProxy)
