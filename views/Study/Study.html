from js import document, location queries = location.href.split('?')[1] no = queries[len(queries) - 1] videoFilePath = '/videos/video_1_{}.mp4'.format(no) console.log(videoFilePath) videoTag = document.getElementById('studyVideo') source = document.createElement('source') source.setAttribute('src', videoFilePath) source.setAttribute('type', 'video/mp4') videoTag.appendChild(source)
